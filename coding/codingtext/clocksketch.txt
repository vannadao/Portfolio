function Flower(){
  beginShape();
  for (var a = 0; a < TWO_PI; a += 0.4) {
    var r = 200 * cos( (6/2) * (5*a));
    var x = r * (cos(a));
    var y = (10*r) * (sin(a));
    vertex(x,y);
    strokeWeight(1);
    noFill();

  }
  endShape(CLOSE);
}
